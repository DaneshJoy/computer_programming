ساختار اولیه کلاس

کلاس = نقشه، شیء = خانه‌ای که از روی نقشه ساخته می‌شود

📋
Class مثل فرم خالی پرونده بیمار است — ساختار را تعریف می‌کند.
Object مثل پرونده پُر شده یک بیمار خاص است — داده واقعی دارد.
از یک فرم می‌توان هزاران پرونده ساخت!
patient.py 
# تعریف کلاس — نقشه بیمار
class Patient:
    pass   # فعلاً خالی — فقط ساختار

# ساخت دو شیء (Object) از روی این کلاس
patient1 = Patient()
patient2 = Patient()

print(type(patient1))   # <class '__main__.Patient'>
print(patient1 is patient2)  # False — دو شیء جداگانه

متد __init__ و Attribute‌ها

مقداردهی اولیه — قلب هر کلاس

🔧
__init__ مثل تکمیل فرم پذیرش بیمار است — وقتی بیمار وارد می‌شود، این اطلاعات حتماً ثبت می‌شوند. self یعنی «این بیمار خاص» — نه همه بیماران، بلکه همین یکی که الان داریم ثبتش می‌کنیم.
patient_v2.py 
class Patient:
    def __init__(self, name, age, heart_rate, blood_pressure):
        # self.attribute = مقدار ورودی
        self.name           = name
        self.age            = age
        self.heart_rate     = heart_rate      # bpm
        self.blood_pressure = blood_pressure  # mmHg
        self.status         = "پذیرش شده"     # مقدار پیش‌فرض


# ────── ساخت بیماران ──────
p1 = Patient("علی رضایی", 45, 78, 120)
p2 = Patient("مریم احمدی", 32, 92, 140)

# دسترسی به attribute‌ها
print(p1.name)            # علی رضایی
print(p2.heart_rate)      # 92
print(p1.status)          # پذیرش شده

# تغییر مقدار
p1.status = "بستری"
print(p1.status)          # بستری

self همیشه اولین پارامتر متد است. پایتون آن را خودکار پر می‌کند — موقع فراخوانی نباید برایش مقدار بدهید.

متدها — رفتار کلاس

توابعی که داخل کلاس تعریف می‌شوند و روی داده‌های همان شیء کار می‌کنند

🩺
Attribute‌ها اطلاعات بیمار هستند. متدها کارهایی هستند که می‌توان با آن بیمار انجام داد: چک کردن وضعیت، صدور گزارش، تجویز دارو... متد = تابع اختصاصی هر شیء.

class Patient:
    def __init__(self, name, age, heart_rate, blood_pressure):
        self.name           = name
        self.age            = age
        self.heart_rate     = heart_rate
        self.blood_pressure = blood_pressure

    # ── متد ۱: نمایش اطلاعات ──
    def show_info(self):
        print(f"👤 بیمار: {self.name} | سن: {self.age}")
        print(f"   ضربان قلب: {self.heart_rate} bpm")
        print(f"   فشار خون: {self.blood_pressure} mmHg")

    # ── متد ۲: بررسی وضعیت قلبی ──
    def check_heart_rate(self):
        if self.heart_rate < 60:
            return "⚠️ برادیکاردی — ضربان کم"
        elif self.heart_rate <= 100:
            return "✅ نرمال"
        else:
            return "🚨 تاکیکاردی — ضربان بالا"

    # ── متد ۳: بررسی فشار خون ──
    def check_blood_pressure(self):
        if self.blood_pressure < 90:
            return "⚠️ هیپوتنسیون"
        elif self.blood_pressure <= 130:
            return "✅ نرمال"
        else:
            return "🚨 هیپرتنسیون"

    # ── متد ۴: گزارش کامل ──
    def full_report(self):
        print("=" * 35)
        self.show_info()
        print(f"   وضعیت قلب: {self.check_heart_rate()}")
        print(f"   فشار خون:  {self.check_blood_pressure()}")
        print("=" * 35)


# ────── استفاده ──────

p2 = Patient("مریم احمدی", 32, 115, 155)

p1.full_report()

وراثت (Inheritance)

قدرتمندترین ویژگی OOP — استفاده مجدد از کد

🧬
همه بیماران پرونده دارند، اما بیمار ICU اطلاعات بیشتری نیاز دارد (دستگاه تنفس مصنوعی، سطح اکسیژن...). به جای نوشتن کد از صفر، از کلاس Patient ارث‌بری می‌کنیم و فقط چیز جدید اضافه می‌کنیم.
icu_patient.py 
# ── کلاس پایه (Parent) ──
class Patient:
    def __init__(self, name, age, heart_rate, blood_pressure):
        self.name           = name
        self.age            = age
        self.heart_rate     = heart_rate
        self.blood_pressure = blood_pressure

    def show_info(self):
        print(f"👤 {self.name} | سن: {self.age}")

    def check_heart_rate(self):
        return "✅ نرمال" if 60 <= self.heart_rate <= 100 else "⚠️ غیرنرمال"


# ── کلاس فرزند (Child) — بیمار ICU ──
class ICUPatient(Patient):   # ← از Patient ارث می‌برد

    def __init__(self, name, age, heart_rate, blood_pressure,
                 oxygen_level, on_ventilator):
        # اول __init__ کلاس پایه را صدا بزن
        super().__init__(name, age, heart_rate, blood_pressure)
        # سپس attribute‌های جدید اضافه کن
        self.oxygen_level   = oxygen_level    # SpO2 %
        self.on_ventilator  = on_ventilator   # True/False

    # ── Override — بازنویسی show_info ──
    def show_info(self):
        super().show_info()   # نمایش اطلاعات پایه
        vent = "بله" if self.on_ventilator else "خیر"
        print(f"   🫁 اکسیژن: {self.oxygen_level}% | ونتیلاتور: {vent}")

    # ── متد جدید — فقط برای ICU ──
    def check_oxygen(self):
        if self.oxygen_level < 90:
            return "🚨 هیپوکسی — نیاز فوری به مداخله"
        elif self.oxygen_level < 95:
            return "⚠️ اکسیژن پایین — پایش دقیق"
        else:
            return "✅ اکسیژن نرمال"

    def icu_report(self):
        print("🏥 گزارش ICU", "-" * 25)
        self.show_info()
        print(f"   قلب:    {self.check_heart_rate()}")
        print(f"   اکسیژن: {self.check_oxygen()}")


# ────── استفاده ──────
regular = Patient("سارا کریمی", 28, 72, 115)
icu_p   = ICUPatient("رضا مظاهری", 67, 52, 85, 88, True)

regular.show_info()
print()
icu_p.icu_report()

# ICUPatient از Patient ارث برده:
print(isinstance(icu_p, Patient))      # True
print(isinstance(icu_p, ICUPatient))   # True

پروژه نهایی — سیستم پایش بیمارستان

همه چیز را با هم ترکیب کنیم

hospital_monitor.py 
class Patient:
    def __init__(self, name, age, heart_rate, blood_pressure):
        self.name           = name
        self.age            = age
        self.heart_rate     = heart_rate
        self.blood_pressure = blood_pressure

    def is_critical(self):
        hr_bad = not (60 <= self.heart_rate <= 100)
        bp_bad = not (90 <= self.blood_pressure <= 130)
        return hr_bad or bp_bad

    def __str__(self):   # وقتی print() روی شیء صدا زده می‌شود
        status = "🚨" if self.is_critical() else "✅"
        return f"{status} {self.name} (HR:{self.heart_rate}, BP:{self.blood_pressure})"


class Hospital:
    def __init__(self, name):
        self.name    = name
        self.patients = []   # لیست بیماران

    def admit(self, patient):
        self.patients.append(patient)
        print(f"✔ {patient.name} پذیرش شد")

    def critical_alert(self):
        print(f"\n🏥 {self.name} — بیماران بحرانی:")
        found = False
        for p in self.patients:
            if p.is_critical():
                print(f"  {p}")
                found = True
        if not found:
            print("  همه بیماران وضعیت پایدار دارند ✅")

    def summary(self):
        total    = len(self.patients)
        critical = sum(1 for p in self.patients if p.is_critical())
        print(f"\n📊 {self.name}: {total} بیمار | {critical} بحرانی")


# ────── اجرا ──────
hospital = Hospital("بیمارستان امام خمینی")

hospital.admit(Patient("علی رضایی",    45, 78,  120))
hospital.admit(Patient("مریم احمدی",   32, 115, 155))
hospital.admit(Patient("رضا مظاهری",  67, 52,  85))
hospital.admit(Patient("سارا کریمی",   28, 72,  115))

hospital.summary()
hospital.critical_alert()
